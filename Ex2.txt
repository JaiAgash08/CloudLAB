import ollama

MODEL = "gemma3:1b"
faq = [
    {"q": "What are your customer support hours?", "a": "Our support is available 9 AM–5 PM, Monday–Friday."},
    {"q": "How do I return a product?", "a": "Use our online returns portal within 30 days of delivery."},
    {"q": "Do you offer free shipping?", "a": "We offer free shipping on orders over $50."}
]
def faq_chat(user):
    try:
        prompt = f"You are a FAQ bot. User: {user}\nAnswer:"
        response = ollama.generate(model=MODEL, prompt=prompt)
        return response["response"]
    except Exception as e:
        return f"⚠️ Error talking to Ollama: {e}"

print("Ready.")
while True:
    q = input("You: ")
    if q == "exit": break
    print("Bot:", faq_chat(q))
